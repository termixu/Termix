🛠️ Практика

```cpp
#include <iostream>
#include <string>

class Pirate {
protected:
    std::string name;
public:
    Pirate(const std::string& n) : name(n) {}
    virtual void speak() const {
        std::cout << name << ": Арр!" << std::endl;
    }
    virtual ~Pirate() = default;
};

class Captain : public Pirate {
public:
    Captain(const std::string& n) : Pirate(n) {}
    void speak() const override {
        std::cout << name << ": Бип! Капитан на мостике!" << std::endl;
    }
};

int main() {
    Pirate* crew[2];
    crew[0] = new Pirate("Билл");
    crew[1] = new Captain("Termix");
    
    for (auto* p : crew) p->speak();
    for (auto* p : crew) delete p;
    
    return 0;
}
```

```cpp
#include <iostream>
#include <fstream>
#include <string>

int main() {
    // Запись
    std::ofstream out("logbook.txt", std::ios::app);
    out << "День 1: Нашёл золотой дублон." << std::endl;
    out << "День 2: Шторм." << std::endl;
    out.close();
    
    // Чтение
    std::ifstream in("logbook.txt");
    std::string line;
    while (std::getline(in, line)) {
        std::cout << line << std::endl;
    }
    in.close();
    
    return 0;
}
```

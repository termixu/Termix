## 🧠 ФИЛОСОФИЯ КУРСА

Обычный подход: JavaScript → синтаксис → точка с запятой → скучно → «я не понимаю».

Подход Termix Academy: JavaScript → магия → кот двигается → игра работает → «я хочу ещё!».

JavaScript — это язык, на котором говорит весь интернет. Каждая кнопка, каждая анимация, каждая игра в браузере работает на JS. Termix ведёт юных пиратов от первой переменной до собственной браузерной игры. Без страха. С весельем. С крюком из Enter.


```javascript
let termix = document.querySelector("#termix")
let scoreDisplay = document.querySelector("#score")
let score = 0

termix.addEventListener("click", function() {
    score = score + 1
    scoreDisplay.innerText = score
    moveTermix()
})

termix.addEventListener("mouseover", function() {
    termix.style.transform = "scale(1.2)"
})

termix.addEventListener("mouseout", function() {
    termix.style.transform = "scale(1.0)"
})

document.addEventListener("keydown", function(event) {
    if (event.key === "ArrowRight") {
        termix.style.left = (parseInt(termix.style.left) || 0) + 20 + "px"
    }
    if (event.key === "ArrowLeft") {
        termix.style.left = (parseInt(termix.style.left) || 0) - 20 + "px"
    }
})

function moveTermix() {
    let maxX = 600 - 64
    let maxY = 400 - 64
    let randomX = Math.floor(Math.random() * maxX)
    let randomY = Math.floor(Math.random() * maxY)
    termix.style.left = randomX + "px"
    termix.style.top = randomY + "px"
}

moveTermix()
```

· 0–10 мин: Termix: «Время — деньги! Точнее, рыба. Сегодня я буду мигать, двигаться сам и напоминать о перерыве каждые 20 минут!»

🛠️ Практика

Ребёнок делает Termix'а, который живёт своей жизнью:

```javascript
let termix = document.querySelector("#termix")
let isAwake = true

// Мигание (каждые 2 секунды Termix меняет цвет хвоста)
setInterval(function() {
    if (isAwake) {
        termix.style.borderColor = 
            termix.style.borderColor === "green" ? "yellow" : "green"
    }
}, 2000)

// Авто-движение (каждые 3 секунды Termix сдвигается)
setInterval(function() {
    let currentLeft = parseInt(termix.style.left) || 0
    termix.style.left = currentLeft + 50 + "px"
```

```javascript
let canvas = document.querySelector("#sea")
let ctx = canvas.getContext("2d")

let shipX = 100
let waveOffset = 0

function draw() {
    // Очистка
    ctx.clearRect(0, 0, 600, 400)
    
    // Небо
    ctx.fillStyle = "#87CEEB"
    ctx.fillRect(0, 0, 600, 200)
    
    // Море с волнами
    ctx.fillStyle = "#006994"
    ctx.beginPath()
    ctx.moveTo(0, 200)
    for (let x = 0; x <= 600; x += 20) {
        let y = 200 + Math.sin(x * 0.05 + waveOffset) * 20
        ctx.lineTo(x, y)
    }
    ctx.lineTo(600, 400)
    ctx.lineTo(0, 400)
    ctx.closePath()
    ctx.fill()
    
    // Корабль Termix'а
    ctx.fillStyle = "#8B4513"
    ctx.fillRect(shipX, 170, 60, 20)
    ctx.fillStyle = "#ffffff"
    ctx.fillRect(shipX + 25, 150, 10, 20)
    ctx.fillStyle = "#ff0000"
    ctx.fillRect(shipX + 25, 150, 10, 10)
    
    // Termix на корабле
    ctx.fillStyle = "#ff6b35"
    ctx.fillRect(shipX + 20, 155, 15, 15)
    ctx.fillStyle = "#00ff41"
    ctx.fillRect(shipX + 20, 165, 15, 3)
    
    // Движение
    shipX = shipX + 0.5
    if (shipX > 600) shipX = -60
    waveOffset = waveOffset + 0.05
    
    requestAnimationFrame(draw)
}
```

🛠️ Практика

JavaScript (каркас игры):

```javascript
let canvas = document.querySelector("#game")
let ctx = canvas.getContext("2d")

// Состояние игры
let snake = [
    {x: 200, y: 200},
    {x: 180, y: 200},
    {x: 160, y: 200}
]
let direction = {x: 20, y: 0}
let food = {x: 100, y: 100}
let score = 0
let isGameOver = false

// Управление
document.addEventListener("keydown", function(event) {
    if (event.key === "ArrowUp" && direction.y === 0) {
        direction = {x: 0, y: -20}
    }
    if (event.key === "ArrowDown" && direction.y === 0) {
        direction = {x: 0, y: 20}
    }
    if (event.key === "ArrowLeft" && direction.x === 0) {
        direction = {x: -20, y: 0}
    }
    if (event.key === "ArrowRight" && direction.x === 0) {
        direction = {x: 20, y: 0}
    }
})

function gameLoop() {
    if (isGameOver) return
    
    // Движение
    let head = {x: snake[0].x + direction.x, y: snake[0].y + direction.y}
    
    // Проверка столкновения со стенами
    if (head.x < 0 || head.x >= 600 || head.y < 0 || head.y >= 400) {
        isGameOver = true
        alert(`Бип! Игра окончена. Счёт: ${score}`)
        return
    }
    
    snake.unshift(head)
    
    // Проверка: съели еду?
    if (head.x === food.x && head.y === food.y) {
        score = score + 10
        food = {
            x: Math.floor(Math.random() * 30) * 20,
            y: Math.floor(Math.random() * 20) * 20
        }
    } else {
        snake.pop()
    }
    
    // Отрисовка
    ctx.fillStyle = "#0a0a0a"
    ctx.fillRect(0, 0, 600, 400)
    
    // Змейка
    ctx.fillStyle = "#00ff41"
    snake.forEach(function(segment) {
        ctx.fillRect(segment.x, segment.y, 18, 18)
    })
    
    // Еда (рыбка)
    ctx.fillStyle = "#ffd700"
    ctx.fillRect(food.x, food.y, 18, 18)
    
    // Счёт
    ctx.fillStyle = "#ffffff"
    ctx.font = "20px monospace"
    ctx.fillText(`Рыбок: ${score / 10}`, 10, 30)
    
    setTimeout(gameLoop, 100)
}

gameLoop()
```

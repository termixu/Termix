· 45–65 мин: Практика. Считаем до 100, перебираем список, делаем таймер.

· 80–90 мин: Итоги. Бадж «Повелитель Циклов».

🛠️ Практика

```python
# while
gold = 0
while gold < 100:
    print(f"У меня {gold} дублонов. Мало!")
    gold += 10
print("Ура! 100 дублонов!")

# for с range
for i in range(5):
    print(f"Попугай кричит {i + 1} раз: Арр!")

# Попугай-повторялка
while True:
    phrase = input("Скажи что-нибудь (или 'стоп'): ")
    if phrase == "стоп":
        print("Попугай улетел!")
        break
    print(f"Попугай: {phrase}!")
```

    "карта острова": {"вес": 1, "ценность": 500}
}

· 80–90 мин: Запуск бота. Он живёт! Бадж «Бот-Боцман».

🛠️ Практика

```python
from telegram import Update
from telegram.ext import Application, CommandHandler, MessageHandler, filters

TOKEN = "ТВОЙ_ТОКЕН_ОТ_BOTFATHER"

async def start(update, context):
    await update.message.reply_text("Бип! Привет, юнга! Я — Termix! Готов к приключениям!")

async def help_command(update, context):
    await update.message.reply_text("/start — начать\n/meme — получить мем\n/status — проверить хвост")

async def meme(update, context):
    await update.message.reply_text("🐱 Кот на клавиатуре: 'Я не сплю, я компилируюсь!'")

async def status(update, context):
    await update.message.reply_text("🟢 Хвост зелёный. Система в порядке!")

async def echo(update, context):
    text = update.message.text.lower()
    if "рыба" in text:
        await update.message.reply_text("Рыба? Где? Давай сюда!")
    elif "бип" in text:
        await update.message.reply_text("Бип-бип! На связи!")
    else:
        await update.message.reply_text(f"Ты сказал: {text}. Но я просто кот. Пиши /help!")

app = Application.builder().token(TOKEN).build()
app.add_handler(CommandHandler("start", start))
app.add_handler(CommandHandler("help", help_command))
app.add_handler(CommandHandler("meme", meme))
app.add_handler(CommandHandler("status", status))
app.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, echo))

print("Бип! Termix запущен!")
app.run_polling()
```
